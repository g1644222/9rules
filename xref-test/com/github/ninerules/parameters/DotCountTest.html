15  public class DotCountTest {
16      private DotCount count0 = new DotCount(0);
17      private DotCount count1 = new DotCount(1);
18      private DotCount count2 = new DotCount(2);
19      private DotCount count3 = new DotCount(3);
20  
21      @Test
22      public void testBasic(){
23          assertThat(Parameters.parameter(DotCount.class, StrictLevel.STRICT), is(count1));
24          assertThat(Parameters.parameter(DotCount.class, StrictLevel.GENERAL), is(count2));
25          assertThat(Parameters.parameter(DotCount.class, StrictLevel.ROUGH), is(count3));
26  
27          assertThat(count1, is(not(new Object())));
28  
29          assertThat(count1, is(not(count2)));
30      }
31  
32      @Test
33      public void testIntegerParameter(){
34          assertThat(count3, is(greaterThan(count0)));
35          assertThat(count3, is(greaterThan(count1)));
36          assertThat(count3, is(greaterThan(count2)));
37          assertThat(count3, is(count3));
38  
39          assertThat(count2, is(greaterThan(count0)));
40          assertThat(count2, is(greaterThan(count1)));
41          assertThat(count2, is(count2));
42          assertThat(count2, is(lessThan(count3)));
43  
44          assertThat(count1, is(greaterThan(count0)));
45          assertThat(count1, is(count1));
46          assertThat(count1, is(lessThan(count2)));
47          assertThat(count1, is(lessThan(count3)));
48  
49          assertThat(count0, is(count0));
50          assertThat(count0, is(lessThan(count1)));
51          assertThat(count0, is(lessThan(count1)));
52          assertThat(count0, is(lessThan(count1)));
53      }
54  
55      @Test
56      public void testSorting(){
57          DotCount[] counts = new DotCount[] { count1, count3, count0, count2, };
58          Arrays.sort(counts);
59          assertThat(counts[0], is(count0));
60          assertThat(counts[1], is(count1));
61          assertThat(counts[2], is(count2));
62          assertThat(counts[3], is(count3));
63      }
64  
65      @Test
66      public void testIntegerParameter2(){
67          assertThat(count1.isEqualsTo(count1), is(true));
68          assertThat(count1.isEqualsTo(new DotCount(1)), is(true));
69  
70          assertThat(count1.isEqualsTo(count2), is(false));
71          assertThat(count1.isEqualsTo(new DotCount(3)), is(false));
72      }
