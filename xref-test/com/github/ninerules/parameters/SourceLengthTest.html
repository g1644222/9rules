14  public class SourceLengthTest {
15      private SourceLength length1 = new SourceLength(50);
16      private SourceLength length2 = new SourceLength(70);
17      private SourceLength length3 = new SourceLength(100);
18      private SourceLength length4 = new SourceLength(30);
19  
20      @Test
21      public void testBasic(){
22          assertThat(Parameters.parameter(SourceLength.class, StrictLevel.STRICT),  is(length1));
23          assertThat(Parameters.parameter(SourceLength.class, StrictLevel.GENERAL), is(length2));
24          assertThat(Parameters.parameter(SourceLength.class, StrictLevel.ROUGH),   is(length3));
25  
26          assertThat(length2, is(lessThan(length3)));
27          assertThat(length1, is(greaterThan(length4)));
28          assertThat(length1.compareTo(length1), is(0));
29  
30          assertThat(length1, is(not(length2)));
31          assertThat(length1, is(not(new Object())));
32          assertThat(length1, is(not(new MethodLength(50))));
33      }
34  
35      @Test
36      public void testFormat(){
37          Message message = new Message("line: %d");
38          assertThat(length1.format(message), is("line: 50"));
39          assertThat(length2.format(message), is("line: 70"));
40          assertThat(length3.format(message), is("line: 100"));
41          assertThat(length4.format(message), is("line: 30"));
42      }
