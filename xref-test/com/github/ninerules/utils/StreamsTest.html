6   import java.util.List;
7   import java.util.stream.Collectors;
8   import java.util.stream.Stream;

10  import org.junit.Before;
11  import org.junit.Test;
12  
13  public class StreamsTest {
14      private Stream<String> streamA;
15      private Stream<Integer> streamB;
16  
17      @Before
18      public void setUp(){
19          streamA = Stream.of("a", "b", "c", "d", "e", "f", "g");
20          streamB = Stream.iterate(1,  index -> index + 1);
21      }
22  
23      @Test
24      public void testBasic(){
25          List<Pair<String, Integer>> list = Streams.zip(streamA, streamB)
26                  .collect(Collectors.toList());
27  
28          assertThat(list.size(), is(7));
29          assertThat(list.get(0).left(),  is("a"));
30          assertThat(list.get(0).right(), is(1));
31          assertThat(list.get(1).left(),  is("b"));
32          assertThat(list.get(1).right(), is(2));
33          assertThat(list.get(2).left(),  is("c"));
34          assertThat(list.get(2).right(), is(3));
35          assertThat(list.get(3).left(),  is("d"));
36          assertThat(list.get(3).right(), is(4));
37          assertThat(list.get(4).left(),  is("e"));
38          assertThat(list.get(4).right(), is(5));
39          assertThat(list.get(5).left(),  is("f"));
40          assertThat(list.get(5).right(), is(6));
41          assertThat(list.get(6).left(),  is("g"));
42          assertThat(list.get(6).right(), is(7));
43      }
44  }
