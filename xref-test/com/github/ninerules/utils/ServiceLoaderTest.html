25  public class ServiceLoaderTest {
26      @Test
27      public void testBasic(){
28          ServiceLoader<Validator> loader = new ServiceLoaderBuilder<Validator>().load(Validator.class);
29  
30          List<Class<Validator>> validators = loader.stream()
31                  .collect(Collectors.toList());
32  
33          assertThat(validators.size(), is(9));
34          assertThat(validators.get(0).getName(), is(NoAccessorValidator.class.getName()));
35          assertThat(validators.get(1).getName(), is(NoElseStatementValidator.class.getName()));
36          assertThat(validators.get(2).getName(), is(FieldCountValidator.class.getName()));
37          assertThat(validators.get(3).getName(), is(FirstClassCollectionValidator.class.getName()));
38          assertThat(validators.get(4).getName(), is(IndentLevelValidator.class.getName()));
39          assertThat(validators.get(5).getName(), is(OneDotPerLineValidator.class.getName()));
40          assertThat(validators.get(6).getName(), is(NoPrimitivesValidator.class.getName()));
41          assertThat(validators.get(7).getName(), is(MethodLengthValidator.class.getName()));
42          assertThat(validators.get(8).getName(), is(SourceLengthValidator.class.getName()));
43      }
44  
45      @Test
46      public void testNotExistsType(){

48  
49          Class<LineCount>[] counts = (Class<LineCount>[]) loader.stream().toArray(size -> new Class<?>[size]);
50          assertThat(counts.length, is(0));
51      }
