13  import org.junit.Test;

15  public class FileVisitorTest {
16      private Path basePath = Paths.get("target/test-classes/hello/src/main");
17  
18      @Test
19      public void testBasic() throws IOException{
20          FileVisitor visitor = new FileVisitor(null);
21          Files.walkFileTree(basePath, visitor);
22  
23          com.github.ninerules.traverser.Paths paths = visitor.createPaths();
24          List<Path> list = paths.stream()
25                  .map(path -> basePath.relativize(path))
26                  .sorted().collect(Collectors.toList());
27  
28          assertThat(list.size(), is(6));
29          assertThat(list.get(0), is(Paths.get("java/sample/hello/GodObject.java")));
30          assertThat(list.get(1), is(Paths.get("java/sample/hello/GodObjectButNotTarget.java")));
31          assertThat(list.get(2), is(Paths.get("java/sample/hello/HelloWorld.java")));
32          assertThat(list.get(3), is(Paths.get("java/sample/hello/Launcher.java")));
33          assertThat(list.get(4), is(Paths.get("webapp/WEB-INF/web.xml")));
34          assertThat(list.get(5), is(Paths.get("webapp/index.html")));
35      }
36  }
