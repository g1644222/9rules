12      public void testBasic(){

14  
15          assertThat(cli.level(), is(StrictLevel.GENERAL));
16  
17          assertThat(cli.hasOption(Option.HELP_OPTION),    is(true));
18          assertThat(cli.hasOption(Option.STRICT_OPTION),  is(true));
19          assertThat(cli.hasOption(Option.GENERAL_OPTION), is(true));
20          assertThat(cli.hasOption(Option.ROUGH_OPTION),   is(false));
21  
22          assertThat(cli.printHelpIfSpecified(), is(true));
23  
24          Argument[] args = cli.arguments().toArray(size -> new Argument[size]);
25          assertThat(args.length, is(2));
26  
27          assertThat(args[0], is(new Argument("src/main/java")));
28          assertThat(args[1], is(new Argument("src/test/java")));
29      }
