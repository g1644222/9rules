6   import java.util.stream.Stream;

10  import com.github.ninerules.StrictLevel;

12  public class OptionsTest {
13      @Test
14      public void testBasic(){
15          Options options = new Options(Stream.of("--help", "--rough", "--strict").map(string -> new Option(string)));
16  
17          Option[] array = options.options().toArray(size -> new Option[size]);
18          assertThat(array.length, is(3));
19  
20          assertThat(array[0], is(new Option("--help")));
21          assertThat(array[1], is(new Option("--rough")));
22          assertThat(array[2], is(new Option("--strict")));
23  
24          assertThat(options.level(), is(StrictLevel.ROUGH));
25  
26          assertThat(options.isSpecified(Option.HELP_OPTION), is(true));
27          assertThat(options.isSpecified(Option.STRICT_OPTION), is(true));
28          assertThat(options.isSpecified(Option.GENERAL_OPTION), is(false));
29          assertThat(options.isSpecified(Option.ROUGH_OPTION), is(true));
30      }
