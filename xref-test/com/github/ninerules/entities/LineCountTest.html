13  public class LineCountTest {
14      private LineCount ten;
15      private LineCount ninerules;
16      private LineCount eleven;
17      private LineCount twelve;
18  
19      @Before
20      public void setUp(){
21          ninerules = new LineCount(9);
22          ten = new LineCount(10);
23          eleven = new LineCount(11);
24          twelve = new LineCount(12);
25      }
26  
27      @Test
28      public void testBasic(){
29          assertThat(ten, is(greaterThan(ninerules)));
30          assertThat(ten, is(lessThan(eleven)));
31          assertThat(ten, is(new LineCount(10)));
32  
33          LineCount ten2 = new LineCount(10);
34          assertThat(ten.hashCode(), is(ten2.hashCode()));
35          assertThat(ten, is(not(eleven)));
36          assertThat(ten, is(not(new Object())));
37      }
38  
39      @Test
40      public void testSequent(){
41          assertThat(ten.isSequent(ninerules), is(true));
42          assertThat(ten.isSequent(eleven), is(true));
43          assertThat(ten.isSequent(twelve), is(false));
44  
45          assertThat(ten.difference(twelve), is(new LineCount(2)));
46      }
47  
48      @Test
49      public void testIllegalArgument(){
50          assertThrows(IllegalArgumentException.class, () -> new LineCount(-1));
51      }
