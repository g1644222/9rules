25  public class OneDotPerLineValidatorTest {
26      private static final String FILE_PATH = "src/test/resources/hello/src/main/java/sample/hello/GodObject.java";
27      private Target target;
28  
29      @Before
30      public void setUp(){
31          Path path = Paths.get(FILE_PATH);
32          target = new NineRulesValidator(StrictLevel.STRICT).parse(path);
33      }
34  
35      @Test
36      public void testValidator(){
37          Validator validator = new OneDotPerLineValidator(StrictLevel.STRICT);
38          Results results = target.accept(validator);
39          List<Violation> violations = getViolations(results.violations);
40  
41          ViolationType type = new ViolationType(OneDotPerLineValidator.ONE_DOT, validator.parameter());
42          assertThat(violations.size(), is(2));
43          assertThat(violations.get(0), 
44                  is(new Violation(type, LineCountsBuilder.build(builder -> builder.of(28)))));
45          assertThat(violations.get(1), 
46                  is(new Violation(type, LineCountsBuilder.build(builder -> builder.of(29)))));
47      }
48  
49      @Test
50      public void testParameter(){
51          Validator validator = new OneDotPerLineValidator(StrictLevel.STRICT);
52  
53          assertThat(validator.parameter(), is(DotCount.STRICT_LEVEL));
54      }
55  
56      private List<Violation> getViolations(Map<FileName, List<Violation>> map){
57          FileName key = new FileName(FILE_PATH);
58          return map.get(key);
59      }
