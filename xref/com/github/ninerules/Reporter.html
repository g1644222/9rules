12  public class Reporter {

14  
15      public Reporter() throws IOException{
16          this(new PrintWriter(new OutputStreamWriter(System.out, "utf-8")));
17      }
18  
19      public Reporter(PrintWriter out) throws IOException{
20          this.out = out;
21      }
22  
23      public void report(Results results){
24          results.report(this);
25          out.flush();
26      }
27  
28      public void report(FileName name, List<Violation> list){
29          out.println(name);
30          list.stream()
31          .forEach(item -> out.println(item));
32      }
