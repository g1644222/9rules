3   import java.util.Optional;

5   import org.eclipse.jdt.core.dom.Block;
6   import org.eclipse.jdt.core.dom.MethodDeclaration;
7   
8   import com.github.ninerules.StrictLevel;
9   import com.github.ninerules.entities.LineCount;
10  import com.github.ninerules.entities.LineCounts;
11  import com.github.ninerules.entities.Message;
12  import com.github.ninerules.parameters.MethodLength;
13  import com.github.ninerules.rules.jdtvisitor.JdtValidator;
14  
15  public class MethodLengthValidator extends JdtValidator {
16      public static final Message TOO_LONG_METHOD = new Message("method is too long (over %s lines).");
17  
18      public MethodLengthValidator(StrictLevel level) {
19          super(level);
20      }
21  
22      @Override
23      public void endVisit(MethodDeclaration node){
24          LineCount difference = counts(node.getBody());
25          checkViolationOfMethodLength(node, difference);
26          super.endVisit(node);
27      }
28  
29      private LineCount counts(Block block){
30          return Optional.ofNullable(block)
31                  .map(item -> super.countLinesOf(item))
32                  .orElse(new LineCount(0));
33      }
34  
35      private void checkViolationOfMethodLength(MethodDeclaration node, LineCount difference){
36          if(difference.isGreaterThan(limitLength())){
37              this.addViolation(buildViolation(TOO_LONG_METHOD, new LineCounts(startLine(node))));
38          }
39      }
40  
41      public LineCount limitLength(){
42          return ((MethodLength)parameter()).convertToLineCount();
43      }
44  
45      @Override
46      public Class<MethodLength> parameterClass() {
47          return MethodLength.class;
48      }
49  }
