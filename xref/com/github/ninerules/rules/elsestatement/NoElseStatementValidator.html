3   import static org.eclipse.jdt.core.dom.ASTNode.IF_STATEMENT;
4   
5   import java.util.Optional;
6   
7   import org.eclipse.jdt.core.dom.IfStatement;
8   import org.eclipse.jdt.core.dom.Statement;
9   
10  import com.github.ninerules.StrictLevel;
11  import com.github.ninerules.entities.LineCounts;
12  import com.github.ninerules.entities.Message;
13  import com.github.ninerules.rules.jdtvisitor.JdtValidator;
14  
15  public class NoElseStatementValidator extends JdtValidator {
16      public static final Message NO_ELSE_STATEMENT = new Message("else statement found.");
17  
18      public NoElseStatementValidator(StrictLevel level) {
19          super(level);
20      }
21  
22      @Override
23      public boolean visit(IfStatement node){
24          checkViolation(node);
25          return super.visit(node);
26      }
27  
28      private void checkViolation(IfStatement node){
29          checkViolation(node, node.getElseStatement());
30      }
31  
32      private void checkViolation(IfStatement node, Statement statement){
33          if(isViolation(Optional.ofNullable(statement))){
34              addViolation(buildViolation(NO_ELSE_STATEMENT, new LineCounts(startLine(node))));
35          }
36      }
37  
38      private boolean isViolation(Optional<Statement> item){
39          return item.map(statement -> {
40              return statement.getNodeType() != IF_STATEMENT;        
41          }).orElse(false);
42      }
43  }
