3   import java.util.function.Function;
4   import java.util.regex.Matcher;
5   import java.util.regex.Pattern;
6   
7   class MatchingCounter implements Function<String, Integer> {
8       private Pattern pattern;
9   
10      public MatchingCounter(Pattern pattern){
11          this.pattern = pattern;
12      }
13  
14      @Override
15      public Integer apply(String string) {
16          return count(pattern.matcher(string));
17      }
18  
19      private int count(Matcher matcher){
20          int count = 0;
21          while(matcher.find()) count++;
22          return count;
23      }
24  }
