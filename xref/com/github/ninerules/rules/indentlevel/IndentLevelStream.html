1   package com.github.ninerules.rules.indentlevel;
2   
3   import java.util.Optional;
4   import java.util.stream.Stream;
5   
6   import com.github.ninerules.parameters.IndentLevel;
7   import com.github.ninerules.utils.StringUtils;
8   
9   public class IndentLevelStream {
10      private IndentLevel buildIndentLevel(String line, String trim) {
11          char firstChar = trim.charAt(0);
12          int indent = line.indexOf(firstChar);
13          return new IndentLevel((indent == 0? 0: indent - 2) / 2);
14      }
15  
16      private IndentLevel buildIndentLevel(String line) {
17          return buildIndentLevel(line, line.trim());
18      }
19  
20      private IndentLevel parseIndentLevel(String line) {
21          Optional<String> indentLevel = StringUtils.optional(line);
22          return indentLevel.map(string -> buildIndentLevel(string))
23                  .orElse(new IndentLevel(0));
24      }
25  
26      public Stream<IndentLevel> stream(IndentManipulator manipulator) {
27          return manipulator.stream()
28                  .map(item -> parseIndentLevel(item))
29                  .distinct();
30      }
31  }
