1   package com.github.ninerules.rules.primitive;
2   
3   import java.util.Arrays;
4   import java.util.List;
5   
6   import org.eclipse.jdt.core.dom.FieldDeclaration;
7   import org.eclipse.jdt.core.dom.Type;
8   
9   class PrimitiveChecker {
10      private List<String> list = Arrays.asList("byte", "short", "int", "long", "float",
11          "double", "boolean", "char", "(java.lang.)?String", "(java.util.)?Date");
12  
13      public boolean check(FieldDeclaration node){
14          Type type = node.getType();
15          return check(type.toString());
16      }
17  
18      private boolean check(String type){
19          return list.stream()
20                  .filter(item -> type.matches(item))
21                  .count() > 0;
22      }    
23  }
