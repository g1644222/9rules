3   import java.util.stream.Stream;
4   
5   import org.eclipse.jdt.core.dom.MethodDeclaration;
6   import org.eclipse.jdt.core.dom.Modifier;
7   
8   class AccessorChecker {
9       public boolean isPublicMethod(MethodDeclaration node){
10          return stream(node)
11                  .filter(modifier -> ((Modifier)modifier).isPublic())
12                  .count() == 1;
13      }
14  
15      @SuppressWarnings("unchecked")
16      private Stream<Object> stream(MethodDeclaration node){
17          return node.modifiers()
18                  .stream()
19                  .filter(modifier -> modifier instanceof Modifier);
20      }
21  }
