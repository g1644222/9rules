3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.eclipse.jdt.core.dom.Type;
7   
8   public class TypeChecker {
9       private static final List<String> COLLECTIONS = new ArrayList<>();
10  
11      static{
12          COLLECTIONS.add("(java.util.)?([A-Z][a-z]+)?List(<.*>)?");
13          COLLECTIONS.add("(java.util.)?([A-Z][a-z]+)?Deque(<.*>)?");
14          COLLECTIONS.add("(java.util.)?([A-Z][a-z]+)?Queue(<.*>)?");
15          COLLECTIONS.add("(java.util.)?([A-Z][a-z]+)?Set(<.*>)?");
16          COLLECTIONS.add("(java.util.)?([A-Z][a-z]+)?Map(<.*, *.*>)?");
17      }
18  
19      public boolean isCollection(Type fieldType){
20          final String type = fieldType.toString();
21          return COLLECTIONS.stream()
22                  .anyMatch(item -> type.matches(item));
23      }
24  }
