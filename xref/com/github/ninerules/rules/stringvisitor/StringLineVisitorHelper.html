3   import java.io.IOException;
4   import java.nio.file.Files;
5   import java.nio.file.Path;
6   import java.util.stream.Stream;
7   
8   import com.github.ninerules.entities.LineCount;
9   import com.github.ninerules.utils.Pair;
10  import com.github.ninerules.utils.Streams;
11  
12  public class StringLineVisitorHelper {
13      private StringLineVisitor visitor;
14  
15      public StringLineVisitorHelper(StringLineVisitor visitor){
16          this.visitor = visitor;
17      }
18  
19      public void visit(Path path){
20          try {
21              visitImpl(path);
22          } catch (IOException e){  }
23      }
24  
25      private void visitImpl(Path path) throws IOException{
26          try(Stream<String> stream = Files.lines(path)){
27              visitLine(stream);
28          }
29      }
30  
31      private void visitLine(Stream<String> stream) throws IOException{
32          Streams.zip(stream, generate())
33          .forEach(pair -> visitLine(pair));
34      }
35  
36      private void visitLine(Pair<String, LineCount> pair){
37          String line = pair.left();
38          LineCount count = pair.right();
39          callVisitorLineInVisitor(line, count);
40      }
41  
42      private void callVisitorLineInVisitor(String line, LineCount count){
43          String convertedLine = visitor.preVisitLine(line, count);
44          visitor.visitLine(convertedLine, count);
45      }
46  
47      private Stream<LineCount> generate(){
48          return Stream.iterate(1, x -> x + 1)
49                  .map(index -> new LineCount(index));        
50      }
51  }
