3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.Optional;
6   import java.util.stream.Stream;
7   
8   import com.github.ninerules.StrictLevel;
9   import com.github.ninerules.Target;
10  import com.github.ninerules.rules.results.Results;
11  import com.github.ninerules.rules.results.ResultsAppender;
12  
13  public class Validators{
14      private List<Validator> list = new ArrayList<>();
15  
16      Validators(){
17      }
18      
19      void register(Validator validator){
20          list.add(validator);
21      }
22  
23      private Stream<Validator> stream(){
24          return list.stream();
25      }
26  
27      public Results validateEach(StrictLevel level, Target unit){
28          return validate(level, unit)
29                  .orElse(Results.empty());
30      }
31      
32      public Optional<Results> validate(StrictLevel level, Target unit){
33          return stream()
34                  .map(checker -> unit.accept(checker))
35                  .reduce((r1, r2) -> new ResultsAppender(r1).append(r2));
36      }
37  }
