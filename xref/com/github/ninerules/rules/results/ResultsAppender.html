1   package com.github.ninerules.rules.results;

3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.Map;
6   
7   import com.github.ninerules.entities.FileName;
8   import com.github.ninerules.rules.violations.Violation;
9   
10  public class ResultsAppender {
11      private Results results;
12  
13      public ResultsAppender(Results results){
14          this.results = results;
15      }
16  
17      public Results append(Results results){
18          results.entryStream()
19          .forEach(item -> put(item));
20          return this.results;
21      }
22  
23      private void put(Map.Entry<FileName, List<Violation>> entry){
24          FileName name = entry.getKey();
25          List<Violation> list = entry.getValue();
26          put(name, list);
27      }
28  
29      private void put(FileName name, List<Violation> list){
30          Map<FileName, List<Violation>> map = results.violations;
31          List<Violation> origList = getMergedList(map, name, list);
32          map.put(name,  origList);
33      }
34  
35      private List<Violation> getMergedList(Map<FileName, List<Violation>> map, FileName name, List<Violation> list){
36          List<Violation> origList = map.getOrDefault(name, new ArrayList<>());
37          origList.addAll(list);
38          return origList;
39      }
40  }
