19      }
20

27      private boolean equals(Parameter parameter){
28          Class<?> clazz1 = getClass();
29          Class<?> clazz2 = parameter.getClass();
30          return Objects.equals(clazz1, clazz2) && isEqualsTo(parameter);
31      }
32  
33      @Override
34      public boolean isEqualsTo(Parameter parameter){
35          return value == ((IntegerParameter)parameter).value;
36      }
37  
38      @Override
39      public boolean isGreaterThan(Parameter parameter){
40          return value > ((IntegerParameter)parameter).value;
41      }
42  
43      @Override
44      public boolean isLessThan(Parameter parameter){
45          return value < ((IntegerParameter)parameter).value;
46      }
