3   import java.util.ArrayList;
4   import java.util.Arrays;
5   import java.util.List;
6   import java.util.function.Consumer;
7   import java.util.stream.IntStream;
8   import java.util.stream.Stream;
9   
10  public class LineCountsBuilder {
11      private List<LineCount> list = new ArrayList<>();
12  
13      private LineCountsBuilder(){
14      }
15  
16      public static LineCounts build(Consumer<LineCountsBuilder> block){
17          LineCountsBuilder builder = new LineCountsBuilder();
18          block.accept(builder);
19          return builder.build();
20      }
21  
22      private LineCounts build(){
23          LineCount[] array = list.stream()
24                  .toArray(size -> new LineCount[size]);
25          return new LineCounts(array);
26      }
27  
28      public LineCountsBuilder stream(Stream<LineCount> stream){
29          stream.forEach(
30                  item -> list.add(item));
31          return this;
32      }
33  
34      public LineCountsBuilder range(int from, int to){
35          IntStream.range(from, to)
36          .forEach(index -> list.add(new LineCount(index)));
37          return this;
38      }
39  
40      public LineCountsBuilder of(int... numbers){
41          Arrays.stream(numbers)
42          .forEach(number -> list.add(new LineCount(number)));
43          return this;
44      }
45  }
