19      public static <T, U> Stream<Pair<T,U>> zip(Stream<T> f, Stream<U> s){
20          Iterator<Pair<T, U>> iterator = iterator(f, s);
21          return StreamSupport.stream(
22                  Spliterators.spliteratorUnknownSize(iterator, NONNULL | ORDERED), false);
23      }
24  
25      private static <T, U> Iterator<Pair<T, U>> iterator(Stream<T> f, Stream<U> s){
26          Iterator<T> fite = f.iterator();
27          Iterator<U> site = s.iterator();
28          return new PairIterator<>(fite, site);
29      }
30  }
