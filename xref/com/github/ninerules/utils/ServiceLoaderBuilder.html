3   import java.io.BufferedReader;
4   import java.io.IOException;
5   import java.io.InputStreamReader;
6   import java.net.URL;
7   import java.util.Optional;
8   import java.util.stream.Stream;
9   
10  public class ServiceLoaderBuilder<T> {
11      public ServiceLoader<T> load(Class<T> clazz){
12          return ExceptionHandler.perform(clazz, 
13                  new ServiceLoader<T>(Stream.empty()),
14                  item -> load(item, item.getClassLoader()));
15      }
16  
17      public ServiceLoader<T> load(Class<T> clazz, ClassLoader loader) throws IOException{
18          String path = "META-INF/services/" + clazz.getName();
19          Optional<URL> url = loadPath(path, loader);
20          return loadFromUrl(url.orElseThrow(() -> new IOException()));
21      }
22  
23      private ServiceLoader<T> loadFromUrl(URL url) throws IOException{
24          try(BufferedReader in = new BufferedReader(new InputStreamReader(url.openStream(), "utf-8"))){
25              return new ServiceLoader<T>(in.lines());
26          }
27      }
28  
29      private Optional<URL> loadPath(String path, ClassLoader loader){
30          URL url = loader.getResource(path);
31          return Optional.ofNullable(url);
32      }
33  }
